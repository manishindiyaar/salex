# Story 1.8: 4-Digit Business Identifier System (WhatsApp Foundation)

## Status
Done

## Story
**As a** salon owner, **I want** to set a unique 4-digit numeric identifier for my business, **so that** customers can easily discover and book with my salon through simple memorable codes that enable WhatsApp smart routing and future discovery features.

## Acceptance Criteria
1. **Routing Code Assignment**: Business owners can set a unique 4-digit numeric code (1000-9999) for their business exactly once
2. **Code Uniqueness**: System prevents duplicate codes and suggests alternatives when preferred codes are taken
3. **Customer Discovery**: Customers can lookup businesses using their 4-digit routing codes through API endpoints
4. **One-Time Setting**: Once set, routing codes cannot be changed to maintain consistency for customer familiarity
5. **Smart Suggestions**: When preferred codes are unavailable, system provides intelligent alternatives using multiple strategies
6. **Performance Requirements**: Code lookup operations must be sub-200ms with 99.9% availability
7. **Range Management**: Support for 9,000 possible codes (1000-9999) with 55% utilization target for regional deployment

### Backend Architecture Implementation [Source: architecture/12-backend-architecture.md]

**Routing Code Service Pattern:**
```typescript
@Injectable()
export class RoutingCodeService {
  constructor(private prisma: PrismaService) {}

  async assignRoutingCode(
    businessId: string,
    ownerId: string,
    preferredCode: string
  ): Promise<RoutingCodeAssignment> {
    // Validate business ownership
    await this.validateBusinessOwnership(businessId, ownerId);
    
    // Check if business already has a routing code (one-time rule)
    const existingBusiness = await this.prisma.business.findUnique({
      where: { id: businessId },
      select: { routing_code: true }
    });
    
    if (existingBusiness?.routing_code) {
      throw new ConflictException('Business already has a routing code assigned. Codes cannot be changed once set.');
    }
    
    // Validate code format (4-digit numeric, 1000-9999)
    if (!this.isValidRoutingCode(preferredCode)) {
      throw new BadRequestException('Routing code must be a 4-digit number between 1000 and 9999');
    }
    
    // Check availability
    const existingCode = await this.prisma.business.findFirst({
      where: { routing_code: preferredCode },
      select: { id: true }
    });
    
    if (existingCode) {
      throw new ConflictException(`Routing code ${preferredCode} is already taken. Use suggestions API for alternatives.`);
    }
    
    // Assign the code with transaction safety
    const updatedBusiness = await this.prisma.business.update({
      where: { id: businessId },
      data: { routing_code: preferredCode },
      select: { id: true, routing_code: true, updatedAt: true }
    });
    
    return {
      success: true,
      routing_code: updatedBusiness.routing_code!,
      businessId: updatedBusiness.id,
      assignedAt: updatedBusiness.updatedAt.toISOString(),
      message: `Routing code ${preferredCode} successfully assigned to your business`
    };
  }

  async suggestAlternativeCodes(preferredCode: string): Promise<RoutingCodeSuggestionsResponse> {
    // Validate preferred code format
    if (!this.isValidRoutingCode(preferredCode)) {
      throw new BadRequestException('Preferred code must be a 4-digit number between 1000 and 9999');
    }
    
    const suggestions: RoutingCodeSuggestion[] = [];
    
    // Strategy 1: Increment Algorithm
    const incrementSuggestions = await this.generateIncrementSuggestions(preferredCode);
    suggestions.push(...incrementSuggestions);
    
    // Strategy 2: Same Prefix Strategy  
    const prefixSuggestions = await this.generateSamePrefixSuggestions(preferredCode);
    suggestions.push(...prefixSuggestions);
    
    // Strategy 3: Random Range Strategy
    const randomSuggestions = await this.generateRandomSuggestions();
    suggestions.push(...randomSuggestions);
    
    // Return top 5 suggestions, prioritizing increment strategy
    const topSuggestions = suggestions
      .filter(s => s.available)
      .slice(0, 5);
    
    return {
      preferred_code: preferredCode,
      suggestions: topSuggestions,
      total_available: await this.countAvailableCodes()
    };
  }

  async lookupBusinessByCode(routingCode: string): Promise<BusinessLookupByCode> {
    // Validate code format
    if (!this.isValidRoutingCode(routingCode)) {
      throw new BadRequestException('Invalid routing code format');
    }
    
    // Optimized lookup with sub-200ms target
    const business = await this.prisma.business.findFirst({
      where: { routing_code: routingCode },
      select: {
        id: true,
        name: true,
        businessType: true,
        phoneNumber: true,
        address: true,
        hoursOfOperation: true
      }
    });
    
    if (!business) {
      return {
        routing_code: routingCode,
        business: {} as any,
        found: false
      };
    }
    
    return {
      routing_code: routingCode,
      business: {
        id: business.id,
        name: business.name,
        businessType: business.businessType,
        phoneNumber: business.phoneNumber,
        address: business.address,
        hoursOfOperation: business.hoursOfOperation as BusinessHours
      },
      found: true
    };
  }

  private isValidRoutingCode(code: string): boolean {
    return /^[1-9][0-9]{3}$/.test(code) && 
           parseInt(code) >= 1000 && 
           parseInt(code) <= 9999;
  }

  private async generateIncrementSuggestions(preferredCode: string): Promise<RoutingCodeSuggestion[]> {
    const suggestions: RoutingCodeSuggestion[] = [];
    const baseCode = parseInt(preferredCode);
    
    for (let i = 1; i <= 10; i++) {
      const candidateCode = (baseCode + i).toString();
      if (candidateCode.length === 4 && parseInt(candidateCode) <= 9999) {
        const available = await this.isCodeAvailable(candidateCode);
        suggestions.push({
          code: candidateCode,
          strategy: 'increment',
          available
        });
        if (suggestions.filter(s => s.available).length >= 3) break;
      }
    }
    
    return suggestions;
  }

  private async generateSamePrefixSuggestions(preferredCode: string): Promise<RoutingCodeSuggestion[]> {
    const suggestions: RoutingCodeSuggestion[] = [];
    const prefix = preferredCode.substring(0, 2);
    
    for (let i = 0; i <= 99; i++) {
      const candidateCode = prefix + i.toString().padStart(2, '0');
      if (candidateCode !== preferredCode) {
        const available = await this.isCodeAvailable(candidateCode);
        suggestions.push({
          code: candidateCode,
          strategy: 'same_prefix',
          available
        });
        if (suggestions.filter(s => s.available).length >= 2) break;
      }
    }
    
    return suggestions;
  }

  private async generateRandomSuggestions(): Promise<RoutingCodeSuggestion[]> {
    const suggestions: RoutingCodeSuggestion[] = [];
    const attempts = 20; // Try 20 random codes
    
    for (let i = 0; i < attempts; i++) {
      const randomCode = (Math.floor(Math.random() * 9000) + 1000).toString();
      const available = await this.isCodeAvailable(randomCode);
      
      suggestions.push({
        code: randomCode,
        strategy: 'random_range',
        available
      });
      
      if (suggestions.filter(s => s.available).length >= 3) break;
    }
    
    return suggestions;
  }

  private async isCodeAvailable(code: string): Promise<boolean> {
    const existing = await this.prisma.business.findFirst({
      where: { routing_code: code },
      select: { id: true }
    });
    return !existing;
  }

  private async countAvailableCodes(): Promise<number> {
    const usedCodes = await this.prisma.business.count({
      where: { routing_code: { not: null } }
    });
    return 9000 - usedCodes; // Total possible codes (1000-9999) minus used codes
  }

  private async validateBusinessOwnership(businessId: string, ownerId: string): Promise<void> {
    const business = await this.prisma.business.findFirst({
      where: { id: businessId, ownerId },
      select: { id: true }
    });
    
    if (!business) {
      throw new ForbiddenException('Access denied: Business not found or not owned by user');
    }
  }
}

@Controller('api/v1')
@UseGuards(ClerkAuthGuard)
export class RoutingCodeController {
  constructor(private routingCodeService: RoutingCodeService) {}

  @Put('businesses/:businessId/routing-code')
  async assignRoutingCode(
    @Param('businessId') businessId: string,
    @Body() dto: AssignRoutingCodeDto,
    @Req() req: any
  ): Promise<ApiResponse<RoutingCodeAssignment>> {
    const result = await this.routingCodeService.assignRoutingCode(
      businessId,
      req.user.id,
      dto.routing_code
    );
    
    return {
      success: true,
      data: result,
      message: 'Routing code assigned successfully'
    };
  }

  @Get('routing-codes/suggestions')
  async getCodeSuggestions(
    @Query('preferred') preferredCode: string
  ): Promise<ApiResponse<RoutingCodeSuggestionsResponse>> {
    const suggestions = await this.routingCodeService.suggestAlternativeCodes(preferredCode);
    
    return {
      success: true,
      data: suggestions,
      message: 'Alternative routing codes generated'
    };
  }

  @Get('businesses/by-code/:routingCode')
  async lookupBusinessByCode(
    @Param('routingCode') routingCode: string
  ): Promise<ApiResponse<BusinessLookupByCode>> {
    const result = await this.routingCodeService.lookupBusinessByCode(routingCode);
    
    return {
      success: true,
      data: result,
      message: result.found ? 'Business found' : 'No business found with this routing code'
    };
  }
}
```
